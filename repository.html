1. Khamoshiyan Text Animation 

import time 
import sys

def print_lyrics():
    lyrics=[
        "Khamoshiyan.....ek saaz hai",
        "Tum dhun koi laao zaraa...",
        "Khamoshiyan........alfaaz hain",
        "Kabhi aa gunguna le zara",
        "Beqarar hain.......baat karne ko",
        "Kehne do inko zaraa haa.........",
        "Khamoshiyan......."
    ]
    
    delays=[1.2, 1.8 ,1.8,2.0, 1.3,2.0, 1.3,2.0,1.0]

    print("khamoshiyan:\n")
    time.sleep(1.5)

    for i, line in enumerate(lyrics):
        for char in line:
            sys.stdout.write(char)
            sys.stdout.flush()
            time.sleep(0.11)
        print()
        time.sleep(delays[i])

print_lyrics()

3. The Number Guessing Game 

#GUESSING THE NUMBER GAME 

import random 
print("==========================================================================================================================")
print("HI!, Welcome to number guessing game \n you have 7 chance to guess the number \n Lets starts:- ")

# u="user_giving_the number"
# c="computer_guessing_the_number"
# r="range"
# gc="guess_count"

c=random.randint(1,50)
r=7
gc=0

while gc < r:
    gc +=1
    u=int(input("ENTER A NUMBER: "))

    if u == c:
        print("You won you guess the number .",c)
        break

    elif gc > c:
        print("Sorry")

    elif u > c:
        print("TOO HIGH! try a lower number .")

    elif u < c:
        print("TOO LOW try a higher number .")
